\begin{thm}
  Consider the two points $(1,1)$ and $(4,5)$. %$(3,6)$ and $(6,2)$
  \begin{enumerate}[(a)]
  \item
    Find the distance between these points.
    \vspace{2in}
  \item
    Find the midpoint of the line segment connecting these points.
    \vspace{2in}
  \item
    Recall that the equation of a circle centered at $(h,k)$ of radius $r$ is
    $$(x - h)^2 + (y - k)^2 = r^2.$$
    Using this equation and the previous two parts, give the equation of a circle passing through $(1,1)$ and $(4,5)$. %$(3,6)$ and $(6,2)$.
  \end{enumerate}	
\end{thm}

\begin{thm}[Bonus]
  Given a quadratic equation
  \begin{equation}\label{b.1}
    ax^2 + bx + c = 0
  \end{equation}
  the quadratic formula gives the two solutions
  \begin{equation}\label{b.2}
    x = \frac{-b \pm \sqrt{b^2 - 4ac}}{2a}.
  \end{equation}
  By completing the square in Equation~\eqref{b.1}, derive Formula~\eqref{b.2}.
\end{thm}
