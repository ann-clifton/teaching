\begin{thm}
  Graph the following functions over the interval $(0, 2\pi)$:
  \begin{enumerate}[(a)]
  \item
    $\displaystyle{y = \sin(2x)}$.
  \item
    $\displaystyle{y = \cos\left(\frac{3}{2}x\right)}$.
  \item
    $\displaystyle{y = \sin(\pi x)}$.
  \end{enumerate}
\end{thm}
